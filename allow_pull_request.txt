name: Pull Request Quality Gate

on:
  pull_request:

jobs:
  check_pull_request:
    runs-on: ubuntu-latest
    steps:
      # Paso 1: Checkout del c칩digo
      - name: Checkout code
        uses: actions/checkout@v2

      # Paso 2: An치lisis de SAST (simulado)
      - name: Realizar an치lisis de SAST
        run: echo "Realizando an치lisis de SAST"

      # Paso extra: Verificar archivos en el directorio
      - name: Verificar archivos en el directorio
        run: ls -la
        working-directory: ${{ github.workspace }}

      # Paso 3: Comprobar si se permite el Pull Request
      - name: Comprobar si se permite el Pull Request
        id: check_allow_pull
        run: |
          if [ -f ${{ github.workspace }}/allow_pull_request.txt ]; then
            echo "allow_pull=true" >> $GITHUB_ENV
            echo "Archivo allow_pull_request.txt encontrado"
          else
            echo "allow_pull=false" >> $GITHUB_ENV
            echo "Archivo allow_pull_request.txt NO encontrado"
          fi
        working-directory: ${{ github.workspace }}

      # Paso 4: Permitir la build si allow_pull es True
      - name: Permitir Pull Request
        if: env.allow_pull == 'true'
        run: echo "Permitiendo Pull Request"

      # Paso 5: Impedir la build si allow_pull es False
      - name: Impedir Pull Request
        if: env.allow_pull == 'false'
        run: |
          echo "Build fallida. Impidiendo Pull Request"
          exit 1
      # Paso 6: Guardar artefactos
      - name: Guardar artefactos
        if: always()
        uses: actions/upload-artifact@v3
        with:
          name: pull-request-check
          path: allow_pull_request.txt
